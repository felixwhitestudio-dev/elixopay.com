ทดสอบด้วย Test Keys แล้วเปลี่ยนเป็น Live Keys เมื่อพร้อม

Example: Initialize Elixopay 
// Server-side (Node.js)
const elixopay = require('elixopay');

elixopay.configure({
  secretKey: 'sk_live_YOUR_SECRET_KEY',
  version: '2026-03-01'
});

// Create a payment
const payment = await elixopay.payments.create({
  amount: 1500.00,     // THB
  currency: 'THB',
  description: 'Order #1234',
  callback_url: 'https://yoursite.com/webhook'
});
